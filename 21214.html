    <button onclick="play('剪刀')">✂</button>

    <button onclick="play('石頭')">✊</button>

<script>

  let playerScore = 0;

  let computerScore = 0;

  let gameOver = false;

  function play(playerChoice) {

    if (gameOver) return;

    const choices = ["剪刀", "石頭", "布"];

    const emojimap = { "剪刀": "✂", "石頭": "✊", "布": "✋" };

    const computerChoice = choices[Math.floor(Math.random() * 3)];

    let result = "";

    if (playerChoice === computerChoice) {

      result = "平手！";

    } else if (

      (playerChoice === "剪刀" && computerChoice === "布") ||

      (playerChoice === "石頭" && computerChoice === "剪刀") ||

      (playerChoice === "布" && computerChoice === "石頭")

    ) {

      result = "你贏了！🎉";

      playerScore++;

    } else {

      result = "你輸了！😢";

      computerScore++;

    }

    document.getElementById("result").innerHTML =

      `你出：${emojimap[playerChoice]}<br>電腦出：${emojimap[computerChoice]}<br><br><strong>${result}</strong>`;

    document.getElementById("playerScore").textContent = playerScore;

    document.getElementById("computerScore").textContent = computerScore;

    if (playerScore === 10 || computerScore === 10) {

      gameOver = true;

      if (playerScore === 10) {

        document.getElementById("result").innerHTML += `<br><br><strong style='color:#4CAF50; font-size:30px;'>🎉 你獲勝了！</strong>`;

      } else {

        document.getElementById("result").innerHTML += `<br><br><strong style='color:#e53935; font-size:30px;'>😢 電腦獲勝！</strong>`;

      }

      document.getElementById("resetBtn").style.display = "inline-block";

    }

  }

  function resetGame() {

    playerScore = 0;

    computerScore = 0;

    gameOver = false;

    document.getElementById("playerScore").textContent = 0;

    document.getElementById("computerScore").textContent = 0;

    document.getElementById("result").textContent = "請出拳！";

    document.getElementById("resetBtn").style.display = "none";

  }

</script>
